# Layer B — derived signals

def compute_tyg(glucose_mmol, triglycerides_mmol):
    glucose_mg = glucose_mmol * 18.018
    trig_mg = triglycerides_mmol * 88.57
    return math.log((glucose_mg * trig_mg) / 2)


def insulin_resistance_signal(tyg):
    if tyg < 8.31:
        return "optimal"
    elif tyg < 8.52:
        return "suboptimal"
    else:
        return "at_risk"

# Layer C — bundle

def insulin_resistance_bundle(signals):

    if risk == "optimal":

    elif risk == "suboptimal":
        narrative = "Early metabolic stress pattern detected."

    else:
        narrative = "Insulin resistance risk pattern detected."

    return {
        "bundle": "insulin_resistance",
        "risk_level": risk,
        "narrative": narrative
    }
```
